用法：cusedown <action> [--param|-alias [param_value ...]] ...

actions:
    search: 搜索一个模组或整合包
        --name|-s: 要搜索的名称
        --version|-v: 游戏版本

    download: 下载一个整合包
        --path|-p: 保存路径
        --id|-id: 整合包的ID。初次下载时必须指定
        --update-mod|-m: 下载模组时不下载整合包指定的版本，而是下载模组的最新版本
        --cn|-cn: 添加中文支持

    modify: 更新一个整合包
        --path|-p: 保存路径
        --cn|-cn: 添加中文支持
        --add|-a: 向整合包中额外添加模组，参数为一系列空格分隔的项目ID
        --rm|-r: 从整合包中移除指定模组，参数为一系列空格分隔的项目ID
        --keep|-k: 在更新整合包时保持指定模组的版本，参数为一系列空格分隔的项目ID
        --lock|-l: 在更新整合包时锁定指定模组的版本，参数为：项目ID 文件ID

    update: 更新整合包中包含的模组到最新版本，可能会引发兼容性问题，请谨慎使用
        --path|-p: 保存路径
